# Importar las librerías
import tkinter as tk           # Crear la interfaz de usuario
from tkinter import messagebox # Permite mostrar mensajes de error

# Definir función para calcular
def calcular():
    try:
        # Obtener valores ingresados y convertir a float
        n1 = float(num1.get())
        n2 = float(num2.get())
        total = n1 + n2   # Suma de los números

        # Condición para mostrar mensaje según el total
        if total > 10:
            resultado.set(f"Total: {total} - Es mayor a 10")
        else:
            resultado.set(f"Total: {total} - Es menor o igual a 10")
    except ValueError:
        messagebox.showerror("Error", "Ingrese valores válidos")

# Crear ventana principal
ventana = tk.Tk()
ventana.geometry("300x300")
ventana.title("Ejemplo")

# Widgets de entrada
tk.Label(ventana, text="Ingrese el primer número").pack()
num1 = tk.Entry(ventana)
num1.pack()

tk.Label(ventana, text="Ingrese el segundo número").pack()
num2 = tk.Entry(ventana)
num2.pack()

# Botón para calcular
tk.Button(ventana, text="Calcular", command=calcular).pack()

# Label para mostrar el resultado
resultado = tk.StringVar()
tk.Label(ventana, textvariable=resultado).pack()

# Ejecutar la ventana
ventana.mainloop()
